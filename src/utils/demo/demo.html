    { id: 'c', status: '3', count: '46' },
    { id: 'd', status: '1', count: '41' }
];
_util.forEach(source, function(i, item){
    console.log(i);
    console.log(item);  
    // 0
    // { id: 'c', status: '3', count: '46' }
    // 1
    // { id: 'd', status: '1', count: '41' }
});

var source = {
    id: 'c',
    status: '3'
};
_util.forEach(source, function(key, value){
    console.log(key);
    console.log(value);  
    // 'id'
    // 'c'
    // 'status'
    // '3'
});
